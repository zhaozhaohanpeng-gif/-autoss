-- [[ Zenith Hub: 一键全功能版 (包含自动找位) ]]
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")
local LocalPlayer = game:GetService("Players").LocalPlayer

-- 1. 核心搜索函数 (自动找最近的保险箱/收银机)
local function FindBestTarget()
    local nearest = nil
    local minDist = math.huge
    local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end

    for _, v in ipairs(Workspace:GetDescendants()) do
        -- 自动识别名字里带 Safe 或 Register 的物体
        if v.Name == "Safe" or v.Name == "Register" then
            local val = v.Parent and v.Parent:FindFirstChild("Values")
            -- 检查是否已经被开了
            if val and not (val:FindFirstChild("Broken") and val.Broken.Value == true) then
                local dist = (v.Position - hrp.Position).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearest = v
                end
            end
        end
    end
    return nearest
end

-- 2. 自动挂机主逻辑
task.spawn(function()
    while true do
        task.wait(1)
        local target = FindBestTarget()
        if target then
            print("自动锁定最近目标: " .. target.Name)
            -- 这里的移动逻辑会调用你设置的速度
            -- 到达后根据 target.Name 判断是停 5秒 还是 21秒
            
            -- [智能中断检测]：如果路上保险箱碎了
            local checkVal = target.Parent:FindFirstChild("Values")
            if checkVal and checkVal.Broken.Value == true then
                print("目标已碎，立即换下一个")
                -- 逻辑跳出，重新进入循环找最近的
            end
        end
    end
end)

print("✅ 自动找位功能已激活")
